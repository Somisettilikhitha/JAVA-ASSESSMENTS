import java.io.BufferedWriter;
import java.io.File;
import java.io.FileWriter;
import java.io.IOException;
import java.io.BufferedReader;
import java.io.FileReader;

public class FileOperations {

    public static void main(String[] args) {
        // Path to the file
        String filePath = "input.txt";

        // Read and print the file line by line
        readFile(filePath);
    }

    // Method to read and print the file line by line
    public static void readFile(String filePath) {
        try (BufferedReader reader = new BufferedReader(new FileReader(filePath))) {
            String line;
            while ((line = reader.readLine()) != null) {
                System.out.println(line);
            }
        } catch (IOException e) {
            System.out.println("An error occurred while reading the file: " + e.getMessage());
        }
    }
}

public class WriteArrayToFile {

    public static void main(String[] args) {
        // Directly passing the array of strings to the method call
        simulateFileWriting(new String[] {
            "Hello, welcome back.",
            "github users!!!.",
            "Writing to a file in Java is easy!",
        });
    }

    // Method to simulate writing an array of strings to a file by printing to the console
    public static void simulateFileWriting(String[] strings) {
        // Simulate writing each string to a file by printing to the console
        for (String str : strings) {
            System.out.println(str);
        }
    }
}
